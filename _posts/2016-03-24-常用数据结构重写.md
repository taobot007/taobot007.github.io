### Trie 树

Trie树要实现三大功能： 插入,判断能不能找到整个单词, 和判断能不能找到部分单词

```python
class Trie:
    def __init__(self):
        self.root = {}        # 整个trie是dict套dict的结构。例如： {a:{b:{-1:True}, c:{-1:True}}} 意味着这里放了两个单词ab和ac。
        self.word_end = -1

    def insert(self, word):
        curNode = self.root
        for c in word:       # 这是最根本的逻辑，一层层往下，其实很直接。
            if not c in curNode:
                curNode[c] = {}
            curNode = curNode[c]

        curNode[self.word_end] = True

    def search(self, word):
        curNode = self.root
        for c in word:
            if not c in curNode:
                return False
            curNode = curNode[c]

        if self.word_end not in curNode:
            return False

        return True

    def startsWith(self, prefix):
        curNode = self.root
        for c in prefix:
            if not c in curNode:
                return False
            curNode = curNode[c]

        return True
```

### Union Find

```python
class QuickFind(object):

    def __init__(self, n):
        self.count = n
        self.id = range(n)

    def connected(self, p, q):
        if self.find(p) == self.find(q):
            return True
        return False

    def find(self, p):   # 这步是最重要的
        while (p != self.id[p]):
            p = self.id[p]
        return p

    def union(self, p, q):
        idq = self.find(q)
        idp = self.find(p)
        if not self.connected(p,q):
            self.id[idp]=idq
            self.count -=1    
```         

### Dijkstra's Algorithm 从一点到另一点的最短路径问题

设图的表示方法是dict的表示,如下:
```
graph_dict = {
  ‘s1’:[(‘s2’,1),(‘s3’,1),(‘s4’,1)],
  ‘s2’:[(‘s1’,1), (‘s4’,1)],
  ‘s3’:[(‘s1’,1), (‘s4’,1)],
  ‘s4’:[(‘s1’,1),(‘s2’,1),(‘s3’,1)],
  }
```
code:
```python
import heapq
def dijkstra(graph_dict, from_node, to_node):
    cost = -1
    ret_path=[]
    q, visited = [[0,from_node,[]]], set()  #这里使用了heap来保证下次找到的一定没visited过的点中是与原点距离最近的那个.
    while q:
        cost, v1, path = heapq.heappop(q)
        if v1 not in visited:
            visited.add(v1)
            path = [v1] + path
            if v1 == to_node: # Find the to_node!!!
                break;
            for v2,c in graph_dict.get(v1, []):
                if v2 not in visited:
                    heapq.heappush(q, (cost+c, v2, path))

    if len(path)>0:
        ret_path = path[::-1]

    return cost,ret_path
```

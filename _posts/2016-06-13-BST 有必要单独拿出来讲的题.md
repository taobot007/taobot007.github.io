### 1 用recursion和非recursion的方式找出BST中第k大的数值

```
# recursion 很好做,基本就是in order traversal

class Solution:
    def find_kth(self, root, k):
        result = [-1, k]

        self.in_order(root, result)

        return result[0]

    def in_oder(self, node, result):
        if not node:
            return

        if result[-1] == 0:
            return  

        self.in_order(node.left, result)

        cur = node

        result[-1] -= 1

        if result[-1] == 0:
            result[0] = node.value
            return

        self.in_order(node.right, result)

# no recursion 的话比较难一些,我们需要使用stack

class Solution:
    def find_kth(self, root, k):

        count = 0
        stack = []

        cur = root

        while cur:
            stack.append(cur)
            cur = cur.left

        while stack:
            cur = stack.pop()
            count += 1
            if count == k:
                return cur.value
            cur = cur.right
            while cur:
                stack.append(cur)
                cur = cur.left

        return -1
    
```
